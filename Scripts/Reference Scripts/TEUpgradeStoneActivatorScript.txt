short activated

	Messagebox "What would you like to upgrade?" "Crystal Sword"
	
	
	
	if Player.GetItemCount Gold001 > 10000
		set activated to 1
		StartQuest TEUpgradeCrystalSwordQuest
	else
		Message "10,000 Gold is required to upgrade this item."
	endif
	Activate Player 0
end

	if activated == 1
		set activated to 2
	endif
End

	endif
End
